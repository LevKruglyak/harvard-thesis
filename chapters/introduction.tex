\chapter{Introduction}


\begin{conjecture}[Generalized Poincar\'e Conjecture]\label{conj:generalized_poincare} Let $\mathscr{C}$ be either $\Top, \PL,$ or $\Diff$. Any $n$-manifold $M\in \mathscr{C}$ which is homotopy equivalent to $S^n$ is also $\mathscr{C}$-isomorphic to $S^n$.
\end{conjecture}

\begin{definition}
  Let $\Theta_n$ be the group of oriented diffeomorphism classes of manifolds homeomorphic to the sphere $S^n$ under the operation of connected sum.
\end{definition}

\begin{definition}
  A closed, compact, oriented, smooth manifold $M$ is said to be a \defn{homotopy $n$-sphere} if it has the homotopy type of the sphere $S^n$.
\end{definition}

\begin{theorem}
  $\Theta_n$ is isomorphic to the group of $h$-cobordism classes of homotopy $n$-spheres under connected sum.
\end{theorem}
